void readInBytes(File file) {
  this.path = file.path;

  //bytes = await new File(filename).readAsBytes();
  bytes = file.readAsBytesSync();

  //print("Filesize = " + bytes.length.toString());

  //print("4 bytes - The ASCII text string " + String.fromCharCodes(bytes.sublist(0, 4)));
  //print("4 bytes - The file size LESS the size of the RIFF description = " + convertLittleEndianToInt(bytes.sublist(4, 8)).toString());
  //print("4 bytes - The ascii text string WAVE = " + String.fromCharCodes(bytes.sublist(8, 12)));
  //print("4 bytes - The ascii text string fmt = "  + String.fromCharCodes(bytes.sublist(12, 16)));
  //print("4 bytes - Size of WAV section chunk = " + convertLittleEndianToInt(bytes.sublist(16, 20)).toString());

  // WAV type format
  // when 1, then linear quantization
  this.audioFormat = convertLittleEndianToInt(bytes.sublist(20, 22));
  //print("2 bytes - WAV type format = " + this.audioFormat.toString());

  // Mono/stereo flag
  // mono (0x01) or stereo (0x02)
  this.numChannels = convertLittleEndianToInt(bytes.sublist(22, 24));
  //print("2 bytes - Mono/stereo flag = " + this.numChannels.toString());

  // Sample frequency.
  this.sampleRate = convertLittleEndianToInt(bytes.sublist(24, 28));
  //print("4 bytes - Sample frequency = " + this.sampleRate.toString());

  // Audio data rate in bytes/sec.
  this.byteRate = convertLittleEndianToInt(bytes.sublist(28, 32));
  //print("4 bytes - Audio data rate in bytes/sec = " + this.byteRate.toString());

  // The block alignment
  this.blockAlignment = convertLittleEndianToInt(bytes.sublist(32, 34));
  //print("2 bytes - The block alignment = " + this.blockAlignment.toString());

  // The number of bits per sample
  this.bitsPerSample = convertLittleEndianToInt(bytes.sublist(34, 36));
  //print("2 bytes - The number of bits per sample = " + this.bitsPerSample.toString());

  // Check where data starts
  int listSize = 0;
  int dataStart = 44;

  if(String.fromCharCodes(bytes.sublist(36, 40)) == "data"){
    //print("4 bytes - The ascii text string data = " + String.fromCharCodes(bytes.sublist(36, 40)));
  } else if(String.fromCharCodes(bytes.sublist(36, 40)) == "LIST"){
    //print("4 bytes - The ascii text string LIST = " + String.fromCharCodes(bytes.sublist(36, 40)));
    listSize = convertLittleEndianToInt(bytes.sublist(40, 44));
    dataStart += listSize;
  } else {
    print("Something went wrong in reading the wav file");
  }

  this.dataLength =
      convertLittleEndianToInt(bytes.sublist(dataStart + 4, dataStart + 8));
  // print("data size = " + this.dataLength.toString());

  // The complete header
  this.header = bytes.sublist(0, dataStart + 8);

  // the actual audio data
  this.data = bytes.sublist(dataStart + 8);
}

void play (String songPath)

Plays a song from a specific path _songPath.

Implementation

void play(String songPath) async {
  if (Platform.isAndroid || Platform.isIOS) {
    try {
      // Add songPath to player.
      methodChannel.invokeMethod('addSongToPlayer', {"path": songPath});
      // Play.
      methodChannel.invokeMethod('play');
      _state = ComponentState.STARTED;
    } on PlatformException catch (e) {
      print(e);
    }
  }
}

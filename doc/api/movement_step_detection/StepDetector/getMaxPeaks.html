Map<int, double> getMaxPeaks(List<double> data, int windowSize, double h) {
  assert (windowSize > 0);
  assert (h > 0.0);
  int length = data.length;
  Map<int, double> indexedPeaks = new Map();
  List<double> peaksS1 = new List(length);
  for(int i = 0; i<length; i++) {
    int start = max(0, i-windowSize);
    int stop = min(length, i+windowSize+1);
    double element = data.elementAt(i);
    List<double> left = data.sublist(start, i);
    List<double> right = data.sublist(i+1, stop);
    peaksS1[i] = peakS1(element, left, right);
  }
  //print('values: '+peaksS1.toString());
  double m = mean(peaksS1);
  //print('mean: '+m.toString());
  double s = stdv(peaksS1, m);
  //print('stdv: '+s.toString());
  for(int i = 0; i<length; i++) {
    if(peaksS1[i] > 0 && (peaksS1[i]-m) > h*s) {
      indexedPeaks.putIfAbsent(i, () => data.elementAt(i));
    }
  }
  Iterable<MapEntry> entries = indexedPeaks.entries;
  for(int i = 0; i<indexedPeaks.length-1; i++) {
    MapEntry entry = entries.elementAt(i);
    int firstKey = entry.key;
    double firstValue = entry.value;
    for(int j = i+1; j<indexedPeaks.length; j++) {
      MapEntry entry2 = entries.elementAt(j);
      int secondKey = entry2.key;
      if(secondKey-firstKey <= windowSize) {
        double secondValue = entry2.value;
        if(firstValue < secondValue) {
          indexedPeaks.remove(firstKey);
          i--;
          break;
        }
        else {
          indexedPeaks.remove(secondKey);
          j--;
        }
      }
    }
  }
  return indexedPeaks;
}

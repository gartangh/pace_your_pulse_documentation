static Future<bool> write(int id, List<Duration> beats, [int bpm = -1, bool flush = false]) async {
  //print("MMD WRITE!");
  await created;
  Completer completer = new Completer<bool>();
  String songID = id.toString();

  if(currentSong == null){  //if first song -> make new one
    currentSong = new MusicMetaData(songID);
    if(bpm != -1){
      currentSong.bpm = bpm;
      if(bpmLibrary[currentSong.bpm] == null) bpmLibrary[currentSong.bpm] = List<String>();
      print("bpmLibrary");
      print(currentSong.bpm);
      print(currentSong.id);
      bpmLibrary[currentSong.bpm].add(currentSong.id);
    }
  }
  if(currentSong?.id == songID){  //if same song, add the beats
    if(bpm != null) currentSong.bpm = bpm;
    currentSong.addBeats(beats);
  }
  if(currentSong?.id != songID || flush){   //if other song or force-flush, flush it and make new object if other song
    MetadataDatabaseController._flush().then((bool ok){
      if(currentSong?.id != songID){
        currentSong = new MusicMetaData(songID);
        if(bpm != -1){
          currentSong.bpm = bpm;
          if(bpmLibrary[currentSong.bpm] == null) bpmLibrary[currentSong.bpm] = List<String>();
          print("bpmLibrary");
          print(currentSong.bpm);
          print(currentSong.id);
          bpmLibrary[currentSong.bpm].add(currentSong.id);
        }
        currentSong.addBeats(beats);
      }
      completer.complete(ok);
    });
  }else{
    completer.complete(true);
  }


  return completer.future;

}

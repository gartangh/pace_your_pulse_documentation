Future<Map<String, dynamic>> read (int id)

Reads a song ID and returns a map with beats and BPM

Implementation

static Future<Map<String,dynamic>> read(int id) async {
  await created;
  Completer completer = new Completer<Map<String,dynamic>>();
  String songID = id.toString();
  if(cachedSong?.id == songID){
    Map<String, dynamic> map = Map<String, dynamic>();
    map['bpm'] = cachedSong.bpm;
    map['beats'] = cachedSong.beats;
    completer.complete(map);
  }else{
    String filename = songID + ".json";
    File file = new File(_workingDirectory.path + "/" + filename);
    file.exists().then((exist){
      if(exist){
        print("MMD READ SONG: Exists in database!");
        file.readAsString().then((String jsonMetaData){
          print("json meta data");
          print(jsonMetaData);
          MusicMetaData metaData = MusicMetaData.fromJson(jsonDecode(jsonMetaData));
          cachedSong = metaData;
          Map<String, dynamic> map = Map<String, dynamic>();
          map['beats'] = cachedSong.beats;
          map['bpm'] = cachedSong.bpm;
          completer.complete(map);
        });
      }else{  //if the file does not exist, just return empty list
        print("MMD READ SONG: Doesn't exist in database!");
        Map<String, dynamic> map = Map<String, dynamic>();
        map['beats'] = List<List<Duration>>();
        map['bpm'] = -1;
        completer.complete(map);
      }
    });
  }
  return completer.future;
}

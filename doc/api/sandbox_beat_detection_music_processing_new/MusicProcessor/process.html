void process(AudioFile aF, String path, Duration length) async {
  Duration beatOffset = Duration(seconds: 0);
  int lengthSeconds = length.inSeconds;
  int frequency = aF.sampleRate;
  int bufferSize = (frequency / window).floor();
  List<int> allData = getChannel(aF.data, 0);

  for(int i=0;i<lengthSeconds;i+=historySeconds){
    if((i+1)*frequency < allData.length) {
      // Get next 30 seconds of song
      List<int> x = allData.sublist(i * frequency, (i + 1) * frequency);

      // Analyze current 30 seconds, with history of 24 seconds
      List beats = analyze(x, frequency, bufferSize, beatOffset);

      // Send beats to other components
      // _musicController.handleNewBeats(beats, beatOffset, path);
      _musicController.processNewBeats(beats, true);

      // Get new beatOffset for next iteration = time between last beat and batchSeconds
      beatOffset = getNextBeatOffset(beats, beatOffset);
    }
  }
}

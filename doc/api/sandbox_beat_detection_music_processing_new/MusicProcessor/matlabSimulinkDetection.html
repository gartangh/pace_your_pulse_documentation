List matlabSimulinkDetection (int bufferSize, List<int> x, int frequency, Duration beatOffset)

                 **********************************************************************************

Dus ik heb aangenomen dat een liedje minstens 50 bpm. Maar voor een moving window, is het wel interessant om de window grootte te hebben als de afstand tussen 2 pieken/beats. Dus bij 50 bpm, zal die afstand het grootste zijn. Dus 50 bpm, zal 6/5 = 1.2 s tussen elke beat. Nu weet je dat je aan 44100 Hz sampled. Maar je sample de samples om de 1024. Dus is de sample-rate in je analyse 44100/1024 = 43,06 Hz. Dus je wilt nu 1,2s muziek analyseren met f = 43,06, wat geeft samples = 1,2 * 43,06 = 51,67 of afgerond 50 samples

List matlabSimulinkDetection(int bufferSize, List<int> x, int frequency, Duration beatOffset) {
  List<List<double>> vectorHistory = initHistory(bufferSize, 1);
  List<double> previousMagnitudes;

  //List<List<bool>> beats = new List();
  int counter = 0;

  /// Fill up history for length of song (in this case 5*bufferSize, so 5 seconds) with
  for (int i = 0; i < x.length && counter < bufferSize; i += window) {
    if (i + window < x.length) {
      // Get current amplitudes
      List<double> amplitudesNormalized = getNormalizedAmplitudes(x.sublist(i, i + window));
      if(i==0) previousMagnitudes = amplitudesNormalized;
      else {
        /// Get current feature vectors
        List<double> vectors = List();
        // j<window
        for (int j = 0; j < 1; j++) vectors.add(pow(amplitudesNormalized[j] - previousMagnitudes[j], 2));

        /// Update previous magnitudes
        previousMagnitudes = amplitudesNormalized;

        /// Update history
        vectorHistory = rightShiftHistory(vectorHistory);
        vectorHistory = updateHistory(vectorHistory, vectors);
      }
      counter++;
    }
  }
  /// For visualisation
  //writeHistory(vectorHistory);

  /// Get average per frequency
  // List<double> currentAvgVectors = getAvgEnergySubbands(vectorHistory, counter);

  /// Get autocorrelation for every frequency
  // Jaime is doing this part
  List<List<int>> autoCor = List();
  for(List<double> vectorsFrequency_i in vectorHistory){
    List<ComplexSimulation> X = fftDouble(vectorsFrequency_i.sublist(0,pow(2,(log(historySeconds*frequency/window)/log(2)).floor())));
    List<ComplexSimulation> conjugatedX = conjugateList(X);
    List<ComplexSimulation> S = dot(X,conjugatedX);
    List<ComplexSimulation> R = ifft(S);
    /// Remove complex 10^(-15) stuff
    List<int> real = getReels(R);
    autoCor.add(real);
  }

  /// For visualisation
  //showAutoCor(autoCor);

  /// Get Best Guess Tempo: get indices of peaks. 43 = 1 seconds. Only for lowest frequency atm. 44100/1024*1 = 43
  // Find peaks in autocorrelation function
  // Distance between peaks e.g. 0.5s => tempo = 60*1/distance = 120
  // Do this for every frequency and take the tempo you think is best. Or only for lowest frequency? Ask Jaime!
  List<int> peaks = findPeaks(autoCor[0]);
  if(peaks != null) {
    int tempo = findTempo(peaks);
    //print("Best guess tempo: " + tempo.toString());

    /// With best guess tempo, use comb filter to find phase
    // Watch video to understand this
    int offset = applyComb(autoCor[0], tempo, frequency);
    //print("offset is: " + offset.toString());

    /// With phase and best guess tempo, find durations
    List<Duration> beats = List();
    beats.add(beatOffset + Duration(milliseconds: (offset / (frequency / window) * 1000).floor()));
    for (int i = 0; i < peaks.length; i++)
      beats.add(Duration(milliseconds: (60 / tempo * 1000).floor()));


    return [null, beats];
  }
  else return null;
}

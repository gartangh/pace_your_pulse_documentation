Updates cadences with cadence for a specific sport

void updateCadences(num cadence, Sport sport){
  int offset = 0;
  if (sport == Sport.CYCLING)
    offset = 5;
  if (cadences[Globals.fitnessModel.selectedZone+offset] == null || cadences[Globals.fitnessModel.selectedZone+offset].isNaN || cadences[Globals.fitnessModel.selectedZone+offset].isInfinite){
    cadences[Globals.fitnessModel.selectedZone+offset] = cadence;
  }
  cadences[Globals.fitnessModel.selectedZone+offset] = cadences[Globals.fitnessModel.selectedZone+offset] + (cadence - cadences[Globals.fitnessModel.selectedZone+offset]
  )/2;
  writeToDatabase();
  print('Cadences updated: $cadences');
}

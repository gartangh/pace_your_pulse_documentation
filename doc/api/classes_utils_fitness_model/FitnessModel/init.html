void init(){
  if (Globals.user.minHeartRate != null &&
      !Globals.user.minHeartRate.isNaN) {
          runningZones[0] =
          (Globals.user.minHeartRate + 0.5 *
                  (Globals.user.maxHeartRate -
                      Globals.user.minHeartRate))*Globals.user.condition;
          runningZones[1] =
              (Globals.user.minHeartRate + 0.6 *
                  (Globals.user.maxHeartRate -
                      Globals.user.minHeartRate))*Globals.user.condition;
          runningZones[2] =
              (Globals.user.minHeartRate + 0.7 *
                  (Globals.user.maxHeartRate -
                      Globals.user.minHeartRate))*Globals.user.condition;
          runningZones[3] =
              Globals.user.minHeartRate + 0.8 *
                  ((Globals.user.maxHeartRate -
                      Globals.user.minHeartRate))*Globals.user.condition;
          runningZones[4] =
              Globals.user.minHeartRate + 0.9 *
                  ((Globals.user.maxHeartRate -
                      Globals.user.minHeartRate))*Globals.user.condition;
          runningZones[5] = Globals.user.maxHeartRate;
    }
  cyclingZones[0] = 0.6 * Globals.user.maxHeartRate*Globals.user.condition;
  cyclingZones[1] = 0.65 * Globals.user.maxHeartRate*Globals.user.condition;
  cyclingZones[2] = 0.75 * Globals.user.maxHeartRate*Globals.user.condition;
  cyclingZones[3] = 0.82 * Globals.user.maxHeartRate*Globals.user.condition;
  cyclingZones[4] = 0.89 * Globals.user.maxHeartRate*Globals.user.condition;
  cyclingZones[5] = 0.94 * Globals.user.maxHeartRate*Globals.user.condition;

  setTargetHR(selectedZone, Sport.RUNNING);
}

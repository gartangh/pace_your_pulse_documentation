void calculateNewSpeed(){         //Kuramoto algorithm
  if(steps.length > 2 && beats.length > 2){
    DateTime now = DateTime.now();

    //finding last 2 beats
    DateTime beat1, beat2;
    int i = 1;
    while(beats[i].isBefore(now))i++;
    if(i<1) i = 2;
    beat1 = beats[i - 2];
    beat2 = beats[i - 1];

    //mean bps music (beat per second)
    double meanBPS = beats.length/(beats.last.difference(beats[0])).inSeconds;

    //last 2 steps
    DateTime step1 = steps[steps.length - 2];
    DateTime step2 = steps[steps.length - 1];

    //Mean steps per second => target BPS for music
    double meanSPS = steps.length/(steps.last.difference(steps[0])).inSeconds;

    double phaseMusic = (now.difference(beat2).abs().inMilliseconds/beat2.difference(beat1).abs().inMilliseconds) * 2 * math.pi;
    double phaseStep = (now.difference(step2).abs().inMilliseconds/step2.difference(step1).abs().inMilliseconds) * 2 * math.pi;
    double diffPhases = phaseMusic - phaseStep;

    var K = 1.3;
    double targetBPS = meanSPS + K/(4 * math.pi) * math.sin(diffPhases);

    double acceleration = targetBPS/meanBPS;

    ChangeCadenceEvent event = new ChangeCadenceEvent(null, acceleration);
    this.sendEvent(event);

    changeMusicTempo(acceleration);
  }
}

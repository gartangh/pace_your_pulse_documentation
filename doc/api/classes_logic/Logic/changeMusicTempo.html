changeMusicTempo method

void changeMusicTempo (double newFactor)

Will adapt all future beats to the new speed factor

Implementation

void changeMusicTempo(double newFactor){
  DateTime now = DateTime.now();
  lastTempoChange = now;
  beats.removeWhere((beat) => beat.isBefore(now));  //remove outdated beats
  lastBeat = now;
  DateTime lastBeatUnModified = now;
  List<DateTime> newBeats = new List<DateTime>();
  for(int i = 0; i<beats.length; i++){
    DateTime beat = beats[i];
    Duration timeBetweenBeats = beat.difference(lastBeatUnModified);
    lastBeatUnModified = beat;
    timeBetweenBeats = new Duration(milliseconds: (timeBetweenBeats.inMilliseconds/newFactor).round());
    lastBeat = lastBeat.add(timeBetweenBeats);
    newBeats.add(lastBeat);
  }
  beats = newBeats;

  this.totalMusicFactorChange *= newFactor;
}

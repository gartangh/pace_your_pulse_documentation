List<int> findPeaks (List<int> autoCor)

Find peaks of autocorrelation function: https://stackoverflow.com/questions/22583391/peak-signal-detection-in-realtime-timeseries-data

Implementation

List<int> findPeaks(List<int> autoCor) {
  //lag 5 for the smoothing functions
  int lag = 50;
  //3.5 standard deviations for signal
  double threshold = 1.645;
  //between 0 and 1, where 1 is normal influence, 0.5 is half
  double influence = 1.0;

  if (autoCor.length <= lag + 2) return null;


  // ********** Initialise variables **********
  List<int> signals = List(autoCor.length);
  List<double> filteredY = List(autoCor.length);
  List<double> avgFilter = List(autoCor.length);
  List<double> stdFilter = List(autoCor.length);
  List<double> subVecStart = List();
  for(int i=0;i<=lag;i++){
    signals[i]= 0;
    avgFilter[i] = 0;
    stdFilter[i] = 0;
    subVecStart.add(autoCor[i] + 0.0);
  }
  for(int i=0;i<autoCor.length;i++) filteredY[i] = autoCor[i] + 0.0;
  signals[lag+1]=0;
  avgFilter[lag] = mean(subVecStart);
  stdFilter[lag] = stdDev(subVecStart, avgFilter[lag]);

  //  ********** RECOGNIZING PEAKS **********
  for (int i = lag + 1; i < autoCor.length; i++) {
    double upper = (autoCor[i] - avgFilter[i - 1]).abs();
    double tr = threshold * stdFilter[i - 1];
    if ( upper > tr ) {
      if (autoCor[i] > avgFilter[i - 1]) {
        signals[i] = 1; //# Positive signal
      }
      else {
        signals[i] = -1; //# Negative signal
      }
      //Make influence lower
      filteredY[i] = influence * autoCor[i] + (1 - influence) * filteredY[i - 1];
    }
    else {
      signals[i] = 0; //# No signal
      filteredY[i] = autoCor[i] + 0.0;
    }
    //Adjust the filters
    List<double> subVec = List();
    for (int j = i - lag; j <= i; j++)
      subVec.add(filteredY[j]);
    avgFilter[i] = mean(subVec);
    stdFilter[i] = stdDev(subVec, avgFilter[i]);
  }


  // ********** GET INDICES OF PEAKS **********
  List<int> indices = List();
  for(int i=0;i<signals.length;i++){
    if(signals[i]==1) indices.add(i);
  }

  // ********** REMOVE PLATEAU **********
  List<int> separate = List();
  for(int i=0;i<indices.length;i++){
    int j=i+1;
    while(j<indices.length && indices[j] == (indices[i] + j-i)) j++;
    int mid = ((i+j)/2).floor();
    separate.add(indices[mid]);
    i = j-1;
  }

  return separate;
}

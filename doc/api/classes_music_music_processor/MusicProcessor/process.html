static Future<void> process(SendPort sendPort) async {

  ReceivePort receivePort = ReceivePort();
  print("Sending sendPort from isolator MusicProcessor!");
  sendPort.send(receivePort.sendPort);



  AudioFile aF;
  int resId;
  Duration length;
  int lengthSeconds;
  Duration beatOffset;
  int frequency;
  int bufferSize;
  List<List<Duration>> databaseBeats;
  int i = 0;
  int bpm;

  List<int> allData;
  receivePort.listen((msg) async {
    switch(msg[0]){
      case 0:
        //print("Received arguments!");
        aF = msg[1];
        resId = msg[2];
        length = msg[3];


        // Offset between end of batchSeconds and last beat. Initially 0
        beatOffset = Duration(seconds: 0);

        // Some handy values
        lengthSeconds = length.inSeconds;
        frequency = aF.sampleRate;
        bufferSize = (frequency / window).floor();
        allData = getChannel(aF.data, aF.numChannels);
        sendPort.send([1, resId]);
        break;
      case 1:
        //print("Received databaseBeats!");
        databaseBeats = msg[1];
        bpm = msg[2];
        //print("Asking to start process");
        sendPort.send([2]);
        break;
      case 2:
        // Start processing Loop
        if(i<lengthSeconds && (i+historySeconds)*frequency < allData.length) {

          // Index of current List<Duration>
          int databaseBeatsIndex = i%historySeconds;
          List res;

          bool inDatabase = !(bpm==-1);
          if(inDatabase)
            res = getBeatsInDatabase(databaseBeats, databaseBeatsIndex, bpm);
          else
            res = getBeatsNotInDatabase(allData, i, frequency, bufferSize, beatOffset);


          List<Duration> beats = res[0];
          int tempo = res[1];

          bool wasFirst = beatOffset == Duration(seconds: 0);

          // Get new beatOffset for next iteration = time between last beat and batchSeconds
          beatOffset = getNextBeatOffset(beats, beatOffset);

          i+=historySeconds;

          _curBPM = tempo;
          print("Current tempo in analyze: " + tempo.toString());
          // Send beats to other components
          //print("Sending beats to other components!");
          sendPort.send([3, inDatabase, resId, beats, wasFirst, tempo]);

        } else
          print("Finished analyzing current song!");
    }
  });


}

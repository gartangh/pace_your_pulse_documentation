fft method

List<Complex> fft (List<Complex> x)

FFT: https://en.wikipedia.org/wiki/Cooley%E2%80%93Tukey_FFT_algorithm

Implementation

static List<Complex> fft(List<Complex> x) {
  var N = x.length;
  List<Complex> X = List(N);
  if (N == 1)
    X[0] = Complex(x[0].real, x[0].complex);
  else {
    List<Complex> even = new List();
    for (int i = 0; i < x.length; i += 2) even.add(x[i]);
    List<Complex> odd = new List();
    for (int i = 1; i < x.length; i += 2) odd.add(x[i]);
    List<Complex> lst1 = fft(even);
    List<Complex> lst2 = fft(odd);
    for (int i = 0; i < N / 2; i++) X[i] = lst1[i];
    for (int i = (N / 2).floor(); i < N; i++)
      X[i] = lst2[i - (N / 2).floor()];
    for (int k = 0; k < N / 2; k++) {
      Complex t = X[k];
      double c = cos(-2 * pi * k / N);
      double q = sin(-2 * pi * k / N);
      Complex n = Complex(
          c * X[k + (N / 2).floor()].real -
              q * X[k + (N / 2).floor()].complex,
          c * X[k + (N / 2).floor()].complex +
              q * X[k + (N / 2).floor()].real);
      X[k] = t + n;
      X[k + (N / 2).floor()] = t - n;
    }
  }
  return X;
}

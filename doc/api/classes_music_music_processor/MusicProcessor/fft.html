List<Complex> fft (List<num> x, int N, int s)

FFT: https://en.wikipedia.org/wiki/Cooley%E2%80%93Tukey_FFT_algorithm

Implementation

List<Complex> fft(List<num> x, int N, int s) {
  List<Complex> X = List(N);
  if (N == 1)
    X[0] = Complex(x[0], 0);
  else {
    List<num> even = new List();
    for (int i = 0; i < x.length; i += 2 * s) even.add(x[i]);
    List<num> odd = new List();
    for (int i = s; i < x.length; i += 2 * s) odd.add(x[i]);
    List<Complex> lst1 = fft(even, (N / 2).floor(), s);
    List<Complex> lst2 = fft(odd, (N / 2).floor(), s);
    for (int i = 0; i < N / 2; i++) X[i] = lst1[i];
    for (int i = (N / 2).floor(); i < N; i++)
      X[i] = lst2[i - (N / 2).floor()];
    for (int k = 0; k < N / 2; k++) {
      Complex t = X[k];
      num c = cos(-2 * pi * k / N);
      num q = sin(-2 * pi * k / N);
      Complex n = Complex(
          c * X[k + (N / 2).floor()].real -
              q * X[k + (N / 2).floor()].complex,
          c * X[k + (N / 2).floor()].complex +
              q * X[k + (N / 2).floor()].real);
      X[k] = t + n;
      X[k + (N / 2).floor()] = t - n;
    }
  }

  return X;
}

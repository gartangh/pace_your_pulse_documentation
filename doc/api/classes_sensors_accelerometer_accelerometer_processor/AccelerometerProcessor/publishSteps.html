void publishSteps() async {
  if (sensor.state == ComponentState.STARTED) {
    if (previousSteps.isNotEmpty) {
      if (updated && previousSteps.length > 1) {
        // Average of 20 last timestamps, maybe add weighted average later
        average = previousSteps.last.difference(previousSteps[0]) ~/
            previousSteps.length;
      }

      updated = false;

      // Publish actual detected steps instead of predicted ones
      while (previousSteps.isNotEmpty &&
          (DateTime.now()
                  .isAfter(previousSteps[0].add(Duration(seconds: 15))) ||
              previousSteps[0].isAfter(DateTime.now()))) {
        previousSteps.removeAt(0);
      }
      if (previousSteps.isNotEmpty) {
        print("STEP:          " + previousSteps[0].toString());
        NextStepEvent event = NextStepEvent(previousSteps[0]);
        Globals.workout.processNextStepEvent(event);
        Globals.logic.processNextStepEvent(event);
        //Globals.eventBroker?.publish(NextStepEvent(previousSteps[0]));

        previousSteps.removeAt(0);
      }
    }
  }
  // If the sensor state is different from the stopped state than we want to keep getting new steps.
  if (sensor.state != ComponentState.STOPPED)
    _publishingThread = Timer(average, () => publishSteps());
}

void getSteps() {
  if (vectorData.isNotEmpty) {
    for (int i = 0; i < vectorData.length; i++) {
      if (vectorData[i] == null || !(vectorData[i] is num)) {
        vectorData.removeAt(i);
        removedData++;
        i--;
      }
    }
    List<num> results = SignalSmoother.signalSmoother.smoothMA(vectorData);
    //print(results);
    Map<num, num> maxPeaks = StepDetector.stepDetector.getMaxPeaks(results, Globals.peakDetectionWindowSize, Globals.maxPeakH);
    Map<num, num> minPeaks = StepDetector.stepDetector.getMinPeaks(
        results, Globals.peakDetectionWindowSize, Globals.minPeakH);

    //print(maxPeaks);
    //print(minPeaks);
    Iterable<MapEntry> tempMax = maxPeaks.entries;
    Iterable<MapEntry> tempMin = minPeaks.entries;

    List<num> peaks = List();
    List<String> check = List();
    int maxIter = 0;
    int minIter = 0;
    String last = 'first';
    for (int i =0;i<tempMin.length + tempMax.length;i++){
      if (maxIter < tempMax.length){
        if (minIter < tempMin.length){
          if (tempMin.elementAt(minIter).key < tempMax.elementAt(maxIter).key){
            if (last == 'first' || last == 'max') {
              peaks.add(tempMin
                  .elementAt(minIter)
                  .key);
              last = 'min';
              check.add(last);
            }
            minIter++;
          }
          else{
            if (last == 'first' || last == 'min'){
              peaks.add(tempMax.elementAt(maxIter).key);
              last = 'max';
              check.add(last);
            }
            maxIter++;
          }
        }
        else{
          if (last == 'first' || last == 'min'){
            peaks.add(tempMax.elementAt(maxIter).key);
            last = 'max';
            check.add(last);
          }
          maxIter++;
        }
      }
      else{
        if (minIter < tempMin.length){
          if (last == 'first' || last == 'max'){
            peaks.add(tempMin.elementAt(minIter).key);
            last = 'min';
            check.add(last);
          }
          minIter++;
        }
      }
    }

    //print(peaks);
    //print(check);
    //print(peaks.length);
    if (check.isNotEmpty) {
      bool minFirst = (check?.elementAt(0) == 'min');
      for (int i = minFirst ? 0 : 1; i < peaks.length; i += 2) {
        previousSteps.add(times?.elementAt(peaks[i]));
      }

      updated = true;
    }
  }
}

void receiveAccelerationDifferent(double acceleration, num realDiffPhases, num meanBPS, num meanSPS) {
  realDiffPhases %= 2 * math.pi;

  double diff = (1-acceleration).abs();
  double speedChange = diff/0.02*4;
  if (!((meanBPS - meanSPS).abs() > 15)) {
    if (realDiffPhases < math.pi / 4 &&
        realDiffPhases > math.pi / 10) { //vertragen
      _currentSPM -= speedChange;
      if (_currentSPM < 100) _currentSPM = 100;
    } else if (realDiffPhases > 3 * math.pi / 4 &&
        realDiffPhases < 19 * math.pi / 10) { //versnellen
      _currentSPM += speedChange;
      if (_currentSPM > 180) _currentSPM = 180;
    }
  }
}

Caches a song when an event is received.

Future cacheSongAfterEvent(msg, String path, f, String title, int resId, Duration length, EventAction type) async {
  a++;
  Globals.tempFileMutex.acquire();
  //print("LOCK LIBRARY " + a.toString() + " ACQUIRED!");
  try {
    if (Platform.isAndroid) {
      List<String> splt = path.split(".");
      String frmat = splt[splt.length - 1];


      File tempFile;
      try {
        tempFile = File('${Globals.tempDirPath}/temp.$frmat');
        if (await tempFile.exists()
        ) {
          tempFile.delete();
          //print("Tempfile LIBRARY " + a.toString() + " deleted!");
        }
        List<num> bytes = await getSongBytes(path);
        await tempFile.writeAsBytes(bytes, flush: true);
        //print("Tempfile LIBRARY " + a.toString() + " witten back!");
        String cachedSongPath = tempFile.uri.toString();
        f([title, path, resId, length, cachedSongPath], type);
      } catch (err) {
        print("ERROR: Could not open the file in Music Library!");
      }
    }
    else if (Platform.isIOS) {
      String cachedSongPath = await _getCacheSong(resId);
      f([title, path, resId, length, cachedSongPath], type);
    }
  } finally {
    //print("LOCK LIBRARY " + a.toString() + " RELEASED!");
    Globals.tempFileMutex.release();
    isolate.kill(priority: Isolate.immediate);
  }
}

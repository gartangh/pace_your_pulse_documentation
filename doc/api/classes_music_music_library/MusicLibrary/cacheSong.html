void cacheSong(Function f, EventAction type, String title, String path, int resId, Duration length) async {
  SendPort sendPort;
  ReceivePort receivePort = ReceivePort();
  receivePort.listen((msg) async {
    if (sendPort == null) {
      print("Receiving sendPort");
      sendPort = msg;
      sendPrt = msg;
    }
    if (Globals.tempFileMutexProceed) cacheSongAfterEvent(msg, path, f, title, resId, length, type);
    else{
      print("LIBRARY SETS PROCEED FALSE");
      Globals.tempFileMutexProceed = false;
    }
    if(msg.runtimeType == bool) cacheSongAfterEvent(msg, path, f, title, resId, length, type);
  });

  if (isolate != null) {
    isolate.kill(priority: Isolate.immediate);
    sendPrt = null;
  }
  isolate = await Isolate.spawn(proceedCheck, receivePort.sendPort);
}

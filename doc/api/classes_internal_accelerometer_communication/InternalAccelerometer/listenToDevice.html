bool listenToDevice (SensorProcessing sensorProcessing)
override

Start listening to accelerometer events.

On every event, add the current timestamp and the accelerometer values to a (separate) local buffer.

Implementation

bool listenToDevice(SensorProcessing sensorProcessing) {
  if (Platform.isIOS || Platform.isAndroid) {
    this.sensorProcessing = sensorProcessing;
    _streamSubscriptions
        .add(accelerometerEvents.listen((AccelerometerEvent event) {
      if (dataBuffer != null && timeStamps != null && event != null) {
        timeStamps.add(DateTime.now());
        //super.dataBuffer.add(event.x);
        //super.dataBuffer.add(event.y);
        //super.dataBuffer.add(event.z);
        //super.dataBuffer.add(
        //    sqrt(event.x * event.x + event.y * event.y + event.z * event.z));
        _rawData[0] = event.x;
        _rawData[1] = event.y;
        _rawData[2] = event.z;
        inputBuffer(_rawData);
      } else
        return false;
    }));
    return true;
  }

  return false;
}

num peakS1(num element, List<num> left, List<num> right) {
  num leftLength = left.length;
  num rightLength = right.length;
  num leftMax = (leftLength > 0) ? -1000000.0 : 0;
  num rightMax = (rightLength > 0) ? -1000000.0 : 0;
  for (int i = 0; i < leftLength; i++) {
    num el = left.elementAt(i);
    num dist = element - el;
    leftMax = max(leftMax, dist);
  }
  for (int i = 0; i < rightLength; i++) {
    num el = right.elementAt(i);
    num dist = element - el;
    rightMax = max(rightMax, dist);
  }
  num avg = 0;
  if (leftLength > 0 && rightLength > 0) {
    avg = (leftMax + rightMax) / 2;
  } else {
    if (leftLength > 0) {
      avg = leftMax;
    } else if (rightLength > 0) avg = rightMax;
  }
  if (avg.abs() > 0.5)
    return avg;
  else
    return 0;
}

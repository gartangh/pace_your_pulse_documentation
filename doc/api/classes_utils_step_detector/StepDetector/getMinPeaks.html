Map<num, num> getMinPeaks(List<num> data, num windowSize, num h) {
  assert(windowSize > 0);
  assert(h > 0.0);
  num length = data.length;
  Map<num, num> indexedPeaks = Map();
  List<num> peaksS1 = List(length);
  for (int i = 0; i < length; i++) {
    num start = max(0, i - windowSize);
    num stop = min(length, i + windowSize + 1);
    num element = data.elementAt(i);
    List<num> left = data.sublist(start, i);
    List<num> right = data.sublist(i + 1, stop);
    peaksS1[i] = peakS1(element, left, right);
  }
  //print('values: '+peaksS1.toString());
  num m = mean(peaksS1);
  //print('mean: '+m.toString());
  num s = stdv(peaksS1, m);
  //print('stdv: '+s.toString());
  for (int i = 0; i < length; i++) {
    if (peaksS1[i] < 0 && (m - peaksS1[i]) > h * s) {
      indexedPeaks.putIfAbsent(i, () => data.elementAt(i));
    }
  }
  Iterable<MapEntry> entries = indexedPeaks.entries;
  for (int i = 0; i < indexedPeaks.length - 1; i++) {
    MapEntry entry = entries.elementAt(i);
    num firstKey = entry.key;
    num firstValue = entry.value;
    for (num j = i + 1; j < indexedPeaks.length; j++) {
      MapEntry entry2 = entries.elementAt(j);
      num secondKey = entry2.key;
      if (secondKey - firstKey <= windowSize) {
        num secondValue = entry2.value;
        if (firstValue > secondValue) {
          indexedPeaks.remove(firstKey);
          i--;
          break;
        } else {
          indexedPeaks.remove(secondKey);
          j--;
        }
      }
    }
  }
  return indexedPeaks;
}

Future writeTimes(String dirPath, List<double> times) async {
  final file = File(dirPath + 'historyTimes.xlsx');
  var bytes = file.readAsBytesSync();
  var decoder = new SpreadsheetDecoder.decodeBytes(bytes, update: true);
  if(times != null) {
    for (int i = 0; i < times.length; i++) {
        decoder.updateCell('Blad1', 0, i, times[i]);
    }
  }

  var newBytes = decoder.encode();
  file.writeAsBytesSync(newBytes);
}

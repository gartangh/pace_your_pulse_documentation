List<double> smoothValuesMA(List<double> input) { //moving average filter
  int length = input.length;
  if (length > 3) {
    List<double> results = new List(length);
    results[0] = (input[0] + input[1] + input[2]) / 3;
    results[1] = (input[0] + input[1] + input[2] + input[3]) / 4;
    for (int i = 2; i < length - 2; i++) {
      results[i] = (input[i - 2] + input[i - 1] + input[i] + input[i + 1] +
          input[i + 2]) / 5;
    }
    results[length - 2] =
        (input[length - 4] + input[length - 3] + input[length - 2] +
            input[length - 1]) / 4;
    results[length - 1] =
        (input[length - 3] + input[length - 2] + input[length - 1]) / 3;
    return results;
  }
  else{
    return new List(0);
  }
}

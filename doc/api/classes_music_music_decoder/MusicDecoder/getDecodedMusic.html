getDecodedMusic(String cachedSongPath, bool inDatabase, Duration start,
    String originalSongPath, Duration length) async {
  _isProcessing = true;
  AudioFile aF = AudioFile();
  _currentStart = start;
  if (inDatabase) {
    // TODO: No decoding needed, just act as a pass-through buffer and give the tuple to the MusicProcessor
  } else {
    // Start Kuramoto loop
    num counter = 0;
    File tempFile;
    while (_isProcessing) {
      if (Platform.isIOS) {
        Directory tempDir = await getTemporaryDirectory();
        tempFile = File('/private' +
            '${tempDir.path}/temp' +
            counter.toString() +
            '.wav');
      } else {
        Directory tempDir = await getApplicationDocumentsDirectory();
        tempFile = File('${tempDir.path}/temp$counter.wav');
      }

      await _ffmpeg.convert(
          cachedSongPath,
          _currentStart.inSeconds.toDouble(),
          _bufferDuration.inSeconds.toDouble(),
          1.0,
          tempFile.path);
      aF.readInBytes(tempFile);

      _musicProcessor.process(
          aF, _currentStart.inMilliseconds == 0, cachedSongPath);
      _currentStart += _bufferDuration;
      counter++;

      if (counter != 0) tempFile.deleteSync();
      if (_currentStart.inMilliseconds > length.inMilliseconds) break;
    }
  }
}
